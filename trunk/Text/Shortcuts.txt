ChromatogramView:

plottingarea:

clicks:
single-click:           select peak/select baseline point/select scan
double-click:           show spectrum
drag:                   zoom in/move baseline point
shift-single-click:     zoom out
shift-double-click:     show all
shift-drag:             move chromatogram
alt-single-click:       select scan
alt-double-click:       add peak
alt-drag:               add custom peak
cmd-single-click:       select scan
cmd-double-click:       add baseline point
cmd-drag:           
alt-cmd-single-click:   select scan
alt-cmd-double-click:   combine spectrum for peak
alt-cmd-drag:           combine spectrum

ctrl-single-click:      contextual menu
                        
scrollwheel:            move chromatogram
alt-scrollwheel:        zoom in
bttn3-single-click:     show all
                        
keys:                   
delete:                 remove peak(s)/remove baseline point(s)
a:                      show all
b:                      show/hide baseline
l:                      show/hide legend
x:                      show/hide axis
g:                      show/hide grid
p:                      show/hide peaks
right-arrow:            select next peak/scan
left-arrow:             select previous peak/scan
shift-right-arrow:      add next peak to selection
shift-left-arrow:       add previous peak to selection
alt-right-arrow:        move chromatogram left
alt-left-arrow:         move chromatogram right
alt-up-arrow:           move chromatogram down
alt-down-arrow:         move chromatogram up
                        
border:                 
drag:                   resize plottingarea
                        
legend:                 
drag:                   move legend
                        
frame:                  
drag:                   zoom with one dimension restricted
